function renderCards(){ const cards = document.getElementById("cards"); const preset = JSON.parse(localStorage.getItem("preset")) || ["g","e","w","q","a"]; const vocab = JSON.parse(localStorage.getItem("vocab")) || {}; cards.innerHTML = ""; preset.forEach(letter=>{ const list = vocab[letter]; if(!list || list.length===0) return; const word = list[0]; // 老師設定的正解 const card = document.createElement("div"); card.className = "card"; card.innerHTML = `
${letter}
${word[0]}
${word[1]}
`; card.onclick = ()=>card.classList.toggle("flip"); cards.appendChild(card); }); }
